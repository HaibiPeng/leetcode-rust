# 思路

1. 遍历 + 栈 翻转；
2. 递归

pub fn reverse(mut head: Option<Box<ListNode>>, mut next_head: Option<Box<ListNode>>) -> Option<Box<ListNode>> {
    while let Some(mut node) = head {
        head = node.next.take();
        node.next = next_head.take();
        next_head = Some(node);
    }
    next_head
